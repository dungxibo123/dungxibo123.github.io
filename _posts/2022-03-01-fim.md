Hàm để chuyển ```transactions``` về dạng đã được sắp xếp theo thứ tự "Lexicography", và lấy được tập $C_{0}$ ban đầu
 * Đầu vào là `transactions` có dạng ```Vector{Vector{String}}```
 * Đầu ra là một `Vector` có 2 phần tử, `transactions` mới và `c0` ($C_0$)
```julia
function transformation(transactions::Vector{Vector{String}})
  new_trans = map((x) -> sort(x,alg=QuickSort), transactions)
  set = Set{Vector{String}}()
  for element in transactions
    map((x) -> push!(set, [x]), element)
  end
  c0 = sort(set_to_vector(set),alg=QuickSort)
  new_trans = sort(new_trans, alg=QuickSort)
  return [new_trans, c0]
end
```
Hàm để chuyển một dữ liệu dạng `Set{T}` về một `Vector{T}` đã được sắp xếp với bất kỳ kiểu dữ liệu `T` nào có operator `<`
```julia
function set_to_vector(set::Set{T}) where T
  res = Vector{T}()
  for element in set
    append!(res,[element])
  end
  return res
end
```
Hàm dùng để đếm xem là một tập các quyết định xuất hiện bao nhiêu lần bên trong `transactions`.
* Đầu vào là 2 tham số, 1 là tập quyết định `ele` có kiểu là `Vector{String}` và tập `transactions` với kiểu là `Vector{Vector{String}}`
* Đầu ra là `count` - số lần xuất hiện của tập quyết định bên trong `transactions` với kiểu dữ liệu là `Int64`
```julia
function count_freq(ele::Vector{String}, transactions::Vector{Vector{String}})
  count = 0::Int64
  map((x) -> count += issubset(ele, x), transactions)
  return count
end
```

Hàm sinh ra `combination` của 2 tập quyết định.
```julia
function unique_combine(v1::Vector{String},v2::Vector{String})
  set = Set{String}()
  temp = Vector{String}()
  append!(temp, v1,v2)
  map((x) -> push!(set,x), temp)
  return sort(set_to_vector(set), alg=QuickSort)
end
```
Hàm sinh tập`C` tương ứng từ tập `F`
```julia
function generate_unique_combination(c::Vector{Vector{String}})
  """
  [["a"],["a","b"],["a", "c"]] -> [["a","b","c"], ["a","c"]]
  """
  if length(c) <= 1
    return Vector{String}()
  end
  set = Set{Vector{String}}()
  for i in range(1,length(c))
    for j in range(i + 1, length(c))
      if issubset(c[i],c[j]) || issubset(c[j],c[i])
        continue
      end
      temp = unique_combine(c[i],c[j]) 
      push!(set, temp)
    end
  end
  combination = sort(set_to_vector(set), alg=QuickSort)
  return combination
end

```

### 2. Thuật toán Projection Tree

#### 2.0 Struct `Node` và các hàm cần thiết
Struct `Node` với 2 thuộc tính,
* `value` là tập quyết định ở `node` đó,
* `childNodes` là một `Dict{String, Node}` để lưu những node con của nó và có thể truy xuất dễ hơn khi sinh tập E của một node nào đó
```julia
struct Node
  value::Vector{String}
  childNodes::Dict{String,Node}
  Node(value::Vector{String}) = new(value, Dict{String,Node}())
end
```
Hàm lấy các keys của một dict và sort nó, để khi duyệt vòng for trong một kết quả được trả về từ hàm này, ta sẽ duyệt các node con của một node theo lexicography
```julia
function dict_sort(keys::Base.KeySet{String, Dict{String, Node}}, alg::Any = QuickSort)
  t = []
  for k in keys
    append!(t,[k])
  end
  return sort(t,alg=alg)
end
```
Hàm để thêm một node con vào một node cha, hàm này có 2 phiên bản, nếu truyền một giá trị con vào thì thêm 1 node, nếu truyền nhiều giá trị con thì thêm tất cả các node tương ứng.

#### 2.1 Mô tả
Là một bản nâng cấp của thuật Apriori, tránh việc sinh ra quá nhiều tập quyết định lặp lại, Projection Tree sử dụng hiệu quả hơn tính lexicography của tập quyết định.


#### 2.2 Thực hiện
Hàm dùng để hoàn thành một node, hàm này là hàm quan trọng nhất, dùng để add tất cả các node con, node cháu. Hàm nhận vào
* Tập `transactions`
* Node `root`, vì node root chứa thông tin về các "1-value" tập quyết định, trong thuật toán Projection Tree, node con của một node $A$ được xây dựng bằng cách lấy tập quyết định của node $A$ đó cộng với các node con của `root` mà có `value` là phần tử cuối của `value` của $A$

```julia
function complete_node(transactions::Vector{Vector{String}}, root::Node, node::Node,results, min_sup = 0.5)
  println("Trying to complete node with value is: ", node.value)
  threshold = Int64(floor(min_sup * length(transactions)))
  for key in dict_sort(keys(root.childNodes))
    println("Trying to complete node with the key value is: ", key)
    if key <= node.value[end]
      continue
    end
    tmp = Vector{String}()
    append!(tmp, node.value, [key])
    println(count_freq(tmp[2:end], transactions),", ", threshold," is two value that need compare")
    if count_freq(tmp[2:end], transactions) > threshold
      println("The $tmp set have freq > threshold, so add it")
      append!(results, [tmp])
      addNode(node, tmp)
    end
  end
  map((x) -> complete_node(transactions, root, x, results,min_sup), [node.childNodes[key]  for key in dict_sort(keys(node.childNodes))])
end
```


Hàm `tree_gen` là hàm chính của bài

```julia
function tree_gen(transactions::Vector{Vector{String}}, c0::Vector{Vector{String}}, results, min_sup::Float64 = 0.5)
  root = Node([""])
  threshold = Int64(floor(min_sup * length(transactions)))
  for ele in c0
    if count_freq(ele, transactions) > threshold
      append!(results, [ele])
      ele = sort(ele,alg=QuickSort)
      addNode(root,ele)
    end
  end
  map((x) -> complete_node(transactions, root, x, results,min_sup), [root.childNodes[key]  for key in dict_sort(keys(root.childNodes))])
  return results
end

```
